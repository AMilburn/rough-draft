Usage

import Tabs from "@/components/Tabs";

<Tabs
  defaultTab="overview"
  tabs={[
    {
      id: "overview",
      label: "Overview",
      content: <p>Content here</p>,
    },
    {
      id: "settings",
      label: "Settings",
      content: <p>Settings here</p>,
      disabled: true,
    },
  ]}
/>
